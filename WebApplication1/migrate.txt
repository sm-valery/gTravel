CREATE PROCEDURE [dbo].[agentSalary]
	@dt1 datetime,
	@dt2 datetime
AS
select ag.AgentId, ag.Name, count(*) sold,
sum(cr.InsPrem) Prem,
sum(cr.InsPremRur) PremRur,
sum(ca.insprem) comm,
sum(ca.inspremrur) commrur,
count(case st.code
when 'annul' then 1 end) annul_count,
sum(case st.code when 'annul' then cr.InsPremRur end) annul_sum_rur ,
sum(case st.code when 'annul' then cr.InsPrem end) annul_sum ,
sum(case st.code when 'annul' then ca.insprem end) annul_comm,
sum(case st.code when 'annul' then ca.inspremrur end) annul_comm_rur
from contract c
join ContractStatus cs on cs.ContractStatusId = c.ContractStatusId
join Status st on st.StatusId = cs.StatusId
join ContractAgent ca on ca.ContractId = c.ContractId
join Agent ag on ag.AgentId = ca.AgentId
join ContractRisk cr on cr.ContractId = c.ContractId
where st.Code in ('confirmed','annul')
and c.date_out between @dt1 and @dt2
group by ag.AgentId, ag.Name
order by ag.Name

CREATE FUNCTION [dbo].[add_str1]
(
	@ContractId uniqueidentifier
)
RETURNS varchar(2000)
AS
BEGIN
	declare @ret varchar(2000)
	set @ret=''

	select @ret = @ret + t.Name +', '
	from dbo.Contract_territory cr
	join dbo.Territory t on t.TerritoryId = cr.TerritoryId
	where cr.ContractId = @ContractId

	return @ret


END
